CREATE TABLE SAMINT.CUSTOMER (
 CNUMBER INT NOT NULL AUTO_INCREMENT,
 CNAME VARCHAR(45), 
 ONAME VARCHAR(75),
 ADDR1 VARCHAR(45), 
 ADDR2 VARCHAR(45),
 POBOX VARCHAR(15), 
 CITY VARCHAR(25), 
 COUNTRY VARCHAR(45),
 CURRENCY VARCHAR(3),
 CONT_PER_NAME VARCHAR(45),
 PHONE_NUMBER INTEGER,
 MAIL_ID VARCHAR(45),
 PRIMARY KEY (CNUMBER)
 );
